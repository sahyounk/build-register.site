import { getWarrantyById } from "/app/nexus.js"; function getWarrantyIdFromUrl() { const params = new URLSearchParams(window.location.search); const id = params.get("id"); if (id && !Number.isNaN(Number(id))) return Number(id); return 1; } function setText(id, value) { const el = document.getElementById(id); if (el) el.textContent = value; } function setBadge(id, text, mode = "success") { const el = document.getElementById(id); if (!el) return; el.textContent = text; el.className = `status-pill ${mode}`; } function shortHash(value) { if (!value || value === "—") return "—"; if (value.length <= 18) return value; return `${value.slice(0, 10)}...${value.slice(-8)}`; } async function loadProperty() { try { const warrantyId = getWarrantyIdFromUrl(); const data = await getWarrantyById(warrantyId); if (!data) { setText("propertyTitle", "Warranty record not found"); setText("propertySubtitle", "No matching on-chain warranty was found for this property view."); setBadge("propertyStatus", "Not Found", "neutral"); return; } const certId = `BR-${String(data.id).padStart(6, "0")}`; setText("propertyTitle", `Property Record • ${certId}`); setText( "propertySubtitle", "Detailed record for handover, warranty coverage, claim activity, and verification status." ); setText("assetId", `Asset ID ${data.assetId}`); setText("completionDate", data.startDate); setText("recordStatus", data.isActive ? "Active Record" : "Inactive Record"); setText("verificationId", certId); setText("auditNote", `Created ${data.createdAt} • Updated ${data.updatedAt}`); setText("warrantyTypePrimary", data.warrantyType || "—"); setText("coveragePrimary", `${data.startDate} → ${data.endDate}`); setText("vendorPrimary", `Vendor ID ${data.vendorId}`); setText("claimState", data.claimStatus || "—"); setText("documentHash", shortHash(data.documentHash || "—")); setText("metadataUri", data.metadataURI || "—"); setBadge("propertyStatus", data.isActive ? "Active Record" : "Inactive", data.isActive ? "success" : "warning"); setBadge("verificationBadge", data.isActive ? "Verified" : "Inactive", data.isActive ? "success" : "warning"); const certLink = document.getElementById("certificateLink"); if (certLink) certLink.href = `/app/certificate.html?id=${data.id}`; } catch (err) { console.error(err); setText("propertyTitle", "Unable to load property record"); setText("propertySubtitle", "The on-chain record could not be loaded at this time."); setBadge("propertyStatus", "Error", "neutral"); } } loadProperty();
